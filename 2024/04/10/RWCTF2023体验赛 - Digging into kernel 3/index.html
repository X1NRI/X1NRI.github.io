__int64 __fastcall rwmod_ioctl(__int64 a1, int a2, request_t *a3)
{
  __int64 v3; // r12
  __int64 idx; // rbx
  __int64 v6; // rdi
  struct request_t kbuf; // [rsp+0h] [rbp-30h] BYREF
  unsigned __int64 v8; // [rsp+18h] [rbp-18h]

  v8 = __readgsqword(0x28u);
  if ( !a3 )
    return -1LL;
  if ( a2 == 0xC0DECAFE )                       // del
  {
    if ( !copy_from_user(&kbuf, a3, 16LL) && kbuf.idx <= 1u )
      kfree(heaparray[kbuf.idx]);               // UAF
    return 0LL;
  }
  v3 = -1LL;
  if ( a2 == 0xDEADBEEF )                       // add
  {
    if ( copy_from_user(&kbuf, a3, 16LL) )
      return 0LL;
    idx = kbuf.idx;
    if ( kbuf.idx > 1u )
      return 0LL;
    heaparray[idx] = _kmalloc(kbuf.size, 3520LL);
    v6 = heaparray[kbuf.idx];
    if ( !v6 )
      return 0LL;
    if ( kbuf.size > 0x7FFFFFFFuLL )
      BUG();
    if ( copy_from_user(v6, kbuf.buf, kbuf.size) )
      return 0LL;
  }
  return v3;
}

最多只能申请两个，只能内核写，存在UAF

有了一个无限大小UAF，我们的利用思路就多种多样了。最大的难点其实在于如何泄露内核基址，因为该内核模块没有调用copy_to_user。

我选择堆喷 user_key_payload 越界读泄露内核基地址

但是有一个问题：add_key()回显分配一个obj拷贝payload后再分配另一个obj作为user_key_payload，如果我们先分配一个obj再释放，调用add_key()则该obj不会直接成为user_key_payload而是在后续数次分配中都作为拷贝payload的临时obj存在。

对于这个题目我们可以直接分配obj1和obj2后全部释放，在obj2上进行UAF就行。

二、EXP

EXP-user_key_payload+seq_operations+pt_regs

有了一个无限大小UAF，我们的利用思路就多种多样了

通过测试：

  • 关闭了 CONFIG_MEMCG_KMEM，这使得GFP_KERNEL 与 GFP_KERNEL_ACCOUNT 会从同样的 kmalloc-xx 中进行分配
  • 关闭了 CONFIG_RANDOMIZE_KSTACK_OFFSET，这使得固定函数调用到内核栈底的偏移值是不变的
  • 关闭了 SLAB_FREELIST_HARDENED，这使得 freelist 几乎没有任何保护，我们可以轻易完成任意地址分配 + 任意地址读写

最大的难点其实在于如何泄露内核基址，因为该内核模块没有调用copy_to_user。

在内核当中存在一个用于密钥管理的子系统，内核提供了 add_key() 系统调用进行密钥的创建，并提供了 keyctl() 系统调用进行密钥的读取、更新、销毁等功能

        #include <sys/types.h>
        #include <keyutils.h>

        key_serial_t add_key(const char *type, const char *description,
                                    const void *payload, size_t plen,
                                    key_serial_t keyring);
//...
       #include <asm/unistd.h>
       #include <linux/keyctl.h>
       #include <unistd.h>

       long syscall(__NR_keyctl, int operation, __kernel_ulong_t arg2,
                    __kernel_ulong_t arg3, __kernel_ulong_t arg4,
                    __kernel_ulong_t arg5);

当我们调用 add_key() 分配一个带有 description 字符串的、类型为 "user" 的、长度为 plen 的内容为 payload 的密钥时，内核会经历如下过程：

  • 首先会在内核空间中分配 obj 1 与 obj2，分配 flag 为 GFP_KERNEL，用以保存 description （字符串，最大大小为 4096）、payload （普通数据，大小无限制）
  • 分配 obj3 保存 description ，分配 obj4 保存 payload，分配 flag 皆为 GFP_KERNEL
  • 释放 obj1 与 obj2，返回密钥 id

其中 obj4 为一个 user_key_payload 结构体，定义如下：

struct user_key_payload {
	struct rcu_head	rcu;		/* RCU destructor */
	unsigned short	datalen;	/* length of this data */
	char		data[] __aligned(__alignof__(u64)); /* actual data */
};

//...

struct callback_head {
	struct callback_head *next;
	void (*func)(struct callback_head *head);
} __attribute__((aligned(sizeof(void *))));
#define rcu_head callback_head

keyctl() 系统调用为我们提供了读取、更新（分配新对象，释放旧对象）、销毁密钥（释放 payload）的功能，其中读取的最大长度由 user_key_payload->datalen 决定，我们不难想到的是我们可以利用题目提供的 UAF 将user_key_payload->datalen 改大，从而完成越界读

注意以下两点：

  • 这里我们的 description 字符串需要和 payload 有着不同的长度，从而简化利用模型
  • 读取 key 时的 len 应当不小于 user_key_payload->datalen，否则会读取失败

add_key()回显分配一个obj拷贝payload后再分配另一个obj作为user_key_payload，如果我们先分配一个obj再释放，调用add_key()则该obj不会直接成为user_key_payload而是在后续数次分配中都作为拷贝payload的临时obj存在。

对于这个题目我们可以直接分配obj1和obj2后全部释放，在obj2上进行UAF就行。然后打seq_operations走pt_regs

EXP-heap spraying+user_key_payload+pipe_buffer

不过除此之外还有一种堆喷的方法能将UAF obj分配到user_key_payload：

  1. 创建UAF obj
  2. 堆喷射 user_key_payload，UAF obj 会一直作为拷贝 payload 的临时 obj 存在。当kmem_cache_cpu的slub page耗光，此时会向 node 的 partial 链表请求新的 slub page 分配user_key_payload，同时原 slub 被加入到 kmem_cache_node 的 full 链表；紧接着UAF obj被释放，该 slub 又从 full 链表迁移到 partial 链表，UAF obj 成为 freelist 的头节点。
  3. 继续堆喷 user_key_payload ，kmem_cache_cpu 的 slub page 耗光，向 node 的 partial 链表请求新的 slub page 分配 user_key_payload。于是UAF obj 所在页面被取回，UAF obj 被分配为 user_key_payload。
